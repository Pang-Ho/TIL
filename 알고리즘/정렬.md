public class Thinking2 {
    public static void main(String[] args) {
        Func func = new Func();
        System.out.println(func.func(6));
    }
}
class Func {
    public int func(int num) {
      if(num < 1) {
        return 0;
      } else if(num == 1) {
        return 1;
      } else if(num == 2) {
        return 2;
      } else if(num == 3) {
        return 4;
      }
    
      return func(num - 1) + func(num - 2) + func(num - 3);
    }
  }
